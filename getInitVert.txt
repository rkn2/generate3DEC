	bi = block_head

	loop while bi # 0
		vi = b_vertex(bi)
		status1 = ''
		loop while vi # 0
			array initialVertex( INSERTarraysize , 1)
			status1 = string(status1) + string(gp_pos(vi))
			;status_BID = string(bi)
